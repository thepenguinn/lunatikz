\documentclass[../_standalone_main]{subfiles}

\ctikzsubcircuitdef{subfigonetriangle} {
    center%
} {
    coordinate (#1-center)
    ++(0,1)
    coordinate (#1-top)
    (#1-center)
    ++(210:1)
    coordinate (#1-left)
    (#1-center)
    ++(-30:1)
    coordinate (#1-right)
    %%
    (#1-top)
    -- (#1-left)
    -- (#1-right)
    -- (#1-top)
    %%
    (#1-left)
    -- ++($(#1-top)-(#1-right)$)
    -- (#1-right)
}

\ctikzsubcircuitactivate{subfigonetriangle}


\ctikzsubcircuitdef{subfigtwotriangles} {
    center%
} {
    \subfigonetriangle {#1-triangle1} {center}
    (#1-triangle1-center)
    ++(1.5,0)
    coordinate (#1-center)
    ++(1.5,0)
    \subfigonetriangle {#1-triangle2} {center}
}

\ctikzsubcircuitactivate{subfigtwotriangles}


\begin{document}

\begin{tikzpicture}
    \draw
    (0,0)
    \subfigonetriangle {triangle} {center}
    ;
\end{tikzpicture}


\begin{tikzpicture}
    \draw
    (0,0)
    \subfigtwotriangles {triangles} {center}
    ;
\end{tikzpicture}


\end{document}
